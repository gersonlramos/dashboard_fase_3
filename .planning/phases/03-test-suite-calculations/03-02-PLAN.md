    - (Saturday, Saturday) -> 0 (or handle via busday semantics)
    - (NaT, date) -> 0
    - (date, NaT) -> 0
    - (fim < inicio) -> 0

    - 'Done' -> contains 'background-color: #90EE90'

    - 'Canceled' -> contains 'background-color: #FFD700'
    - 'Cancelado' -> contains 'background-color: #FFD700'
