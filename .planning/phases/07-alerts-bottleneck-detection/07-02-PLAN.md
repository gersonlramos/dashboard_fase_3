# ── Alertas: velocidade por lake e impedimentos vencidos ─────────────────────

@st.cache_data(ttl=900)
def _calcular_semaforos(df_completo: pd.DataFrame, lakes_df: pd.DataFrame):
    """Wrapper cacheado para calcular_velocidade_necessaria."""
    return calcular_velocidade_necessaria(df_completo, lakes_df)


# Executa calculo (usa o df completo, nao filtrado, para incluir todos os lakes)
_df_semaforos = _calcular_semaforos(df, df_lake)

_lakes_risco   = int((_df_semaforos['semaforo'] == 'risco').sum())
_lakes_atencao = int((_df_semaforos['semaforo'] == 'atencao').sum())

# Impedimentos vencidos: deadline < hoje e status nao concluido
_STATUS_DONE_IMP = {'Done', 'Closed', 'Resolved', 'Concluído', 'Concluida',
                    'Canceled', 'Cancelled', 'Cancelado'}
_csv_imp = os.path.join(DADOS_DIR, 'pendencias_BF3E4-293.csv')
if os.path.exists(_csv_imp):
    _df_imp_alert = pd.read_csv(_csv_imp, encoding='utf-8-sig')
    _df_imp_alert['Deadline'] = pd.to_datetime(
        _df_imp_alert['Deadline'], errors='coerce', dayfirst=True
    )
    _mask_vencido = (
        _df_imp_alert['Deadline'].notna() &
        (_df_imp_alert['Deadline'] < _hoje) &
        (~_df_imp_alert['Status'].isin(_STATUS_DONE_IMP))
    )
    _imp_vencidos = int(_mask_vencido.sum())
else:
    _imp_vencidos = 0

Rules:

- The `_STATUS_DONE_IMP` set intentionally includes accented variants because the
  impedimentos CSV may mix locales.
- The column name in `pendencias_BF3E4-293.csv` is `Deadline` (capital D) — match

- Do NOT wrap the CSV read in `@st.cache_data` directly; the outer wrapper

```python
    # ── Painel de Alertas Automáticos ────────────────────────────────────────
    st.subheader("🚨 Alertas")
    _col_a1, _col_a2, _col_a3 = st.columns(3)

    # Lakes em risco (vermelho)
    with _col_a1:
        if _lakes_risco > 0:
            st.error(f"🔴 **{_lakes_risco}** lake{'s' if _lakes_risco > 1 else ''} em **risco**\n\n"
                     f"Velocidade real < 50% da necessária")
        else:
            st.success("🟢 Nenhum lake em risco de atraso")

    # Lakes em atenção (amarelo)
    with _col_a2:
        if _lakes_atencao > 0:
            st.warning(f"🟡 **{_lakes_atencao}** lake{'s' if _lakes_atencao > 1 else ''} em **atenção**\n\n"
                       f"Velocidade real entre 50–80% da necessária")
        else:
            st.success("🟢 Nenhum lake em atenção")

    # Impedimentos vencidos (badge)
    with _col_a3:
        if _imp_vencidos > 0:
            st.error(f"⚠️ **{_imp_vencidos}** impedimento{'s' if _imp_vencidos > 1 else ''} com **deadline vencida**\n\n"
                     f"Acesse a aba ⚠️ Impedimentos para detalhes")
        else:
            st.success("✅ Nenhum impedimento com deadline vencida")

    st.markdown(hr_style, unsafe_allow_html=True)

    # ── Tabela semáforo por lake ──────────────────────────────────────────────
    if not _df_semaforos.empty:
        st.subheader("🚦 Velocidade por Lake")
        _EMOJI = {'risco': '🔴', 'atencao': '🟡', 'ok': '🟢'}
        _df_sem_display = _df_semaforos.copy()
        _df_sem_display['status'] = _df_sem_display['semaforo'].map(_EMOJI) + ' ' + _df_sem_display['semaforo']
        _df_sem_display['deadline'] = _df_sem_display['deadline'].dt.strftime('%d/%m/%Y')
        _df_sem_display = _df_sem_display.rename(columns={
            'lake': 'Lake',
            'itens_restantes': 'Restantes',
            'dias_uteis': 'Dias Úteis',
            'velocidade_necessaria': 'Vel. Necessária',
            'velocidade_real': 'Vel. Real',
            'ratio': 'Ratio',
            'deadline': 'Deadline',
            'status': 'Semáforo',
        })
        st.dataframe(
            _df_sem_display[['Lake', 'Deadline', 'Dias Úteis', 'Restantes',
                              'Vel. Necessária', 'Vel. Real', 'Ratio', 'Semáforo']],
            use_container_width=True,
            hide_index=True,
        )
        st.markdown(hr_style, unsafe_allow_html=True)
```

Rules:
- Insert the entire block **after** the existing `st.markdown(hr_style, ...)` line

- Use `st.error`, `st.warning`, `st.success` — NOT custom HTML — for the three alert
  boxes. This ensures theme compatibility with both dark and light modes.
- The semaphore table uses `st.dataframe(... hide_index=True)` — no `.style.applymap`
  or `.style.map` — to avoid FutureWarning regression.
- Variable names `_col_a1`, `_col_a2`, `_col_a3` use underscore prefix to avoid
  shadowing the existing `col1`, `col2`, `col3` names used further below.
- Do NOT modify any code outside the Executivo tab block.
