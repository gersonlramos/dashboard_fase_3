def calcular_velocidade_necessaria(df: pd.DataFrame, lakes_fase: pd.DataFrame) -> pd.DataFrame:
    """
    Calcula velocidade necessaria vs. real por lake e classifica o risco.

    Parametros
    ----------
    df : DataFrame com colunas 'Data-Lake', 'Status', 'Data Atualizacao'
         (o df completo nao filtrado, uppercase em Data-Lake).
    lakes_fase : DataFrame com colunas 'lake', 'data_fim'
                 (o df carregado de datas_esperadas_por_lake.csv,
                  com 'lake' em uppercase).

    Retorna
    -------
    DataFrame com colunas:
        lake, itens_restantes, deadline, dias_uteis,
        velocidade_necessaria, velocidade_real, ratio, semaforo
    """
    STATUS_DONE = {
        'done', 'closed', 'resolved',
        'concluído', 'concluida', 'canceled', 'cancelled', 'cancelado',
    }
    hoje = pd.Timestamp.today().normalize()

    # --- deadline por lake (max data_fim) ---
    if lakes_fase.empty or 'lake' not in lakes_fase.columns or 'data_fim' not in lakes_fase.columns:
        return pd.DataFrame(columns=[
            'lake', 'itens_restantes', 'deadline', 'dias_uteis',
            'velocidade_necessaria', 'velocidade_real', 'ratio', 'semaforo',
        ])

    deadlines = (
        lakes_fase
        .dropna(subset=['data_fim'])
        .groupby('lake', as_index=False)['data_fim']
        .max()
        .rename(columns={'data_fim': 'deadline'})
    )

    rows = []
    for _, row in deadlines.iterrows():
        lake = row['lake']
        deadline = pd.Timestamp(row['deadline'])

        mask_lake = df['Data-Lake'].astype(str).str.strip().str.upper() == lake
        df_lake = df[mask_lake].copy()

        status_lower = df_lake['Status'].astype(str).str.strip().str.lower()
        itens_restantes = int((~status_lower.isin(STATUS_DONE)).sum())

        # dias uteis de hoje ate deadline (0 se deadline passou)
        dias_uteis = max(int(np.busday_count(hoje.date(), deadline.date())), 1)

        velocidade_necessaria = itens_restantes / dias_uteis  # itens/dia-util

        # velocidade real: itens concluidos nos ultimos 5 dias uteis / 5
        cinco_du_atras = hoje - pd.offsets.BusinessDay(5)
        mask_recente = (
            status_lower.isin(STATUS_DONE) &
            pd.to_datetime(
                df_lake['Data Atualizacao'], errors='coerce'
            ).between(cinco_du_atras, hoje)
        )
        itens_recentes = int(mask_recente.sum())
        velocidade_real = itens_recentes / 5.0

        # ratio e semaforo
        if velocidade_necessaria <= 0:
            ratio = 1.0
        else:
            ratio = velocidade_real / velocidade_necessaria

        if ratio < 0.5:
            semaforo = 'risco'
        elif ratio < 0.8:
            semaforo = 'atencao'
        else:
            semaforo = 'ok'

        rows.append({
            'lake': lake,
            'itens_restantes': itens_restantes,
            'deadline': deadline,
            'dias_uteis': dias_uteis,
            'velocidade_necessaria': round(velocidade_necessaria, 3),
            'velocidade_real': round(velocidade_real, 3),
            'ratio': round(ratio, 3),
            'semaforo': semaforo,
        })

    return pd.DataFrame(rows)
```

Key rules to follow:
- Use `np.busday_count(hoje.date(), deadline.date())` — both arguments must be
  Python `date` objects, not `pd.Timestamp`. Wrap with `.date()`.
- `max(..., 1)` prevents division by zero when deadline is today or in the past.
- `pd.offsets.BusinessDay(5)` gives the window for real velocity; no calendar-day
  approximation.
- Do NOT import anything new at the top — `numpy` and `pandas` are already imported.
- Do NOT touch any existing function.

# ── TEST: calcular_velocidade_necessaria ─────────────────────────────────────

class TestCalcularVelocidadeNecessaria:
    """Tests for ALERT-04 / ALERT-02 / ALERT-03 requirements."""

    def _make_df(self, rows):
        """Build a minimal subtask DataFrame matching FASE_3.csv schema."""
        return pd.DataFrame(rows, columns=['Data-Lake', 'Status', 'Data Atualizacao'])

    def _make_lakes(self, lake, data_fim):
        return pd.DataFrame({'lake': [lake], 'data_fim': [pd.Timestamp(data_fim)]})

    def test_returns_dataframe_with_expected_columns(self):
        from calculations import calcular_velocidade_necessaria
        df = self._make_df([
            {'Data-Lake': 'COMPRAS', 'Status': 'To Do', 'Data Atualizacao': '2026-03-01'},
        ])
        lakes = self._make_lakes('COMPRAS', '2026-12-31')
        result = calcular_velocidade_necessaria(df, lakes)
        expected_cols = {
            'lake', 'itens_restantes', 'deadline', 'dias_uteis',
            'velocidade_necessaria', 'velocidade_real', 'ratio', 'semaforo',
        }
        assert expected_cols.issubset(set(result.columns))

    def test_done_items_excluded_from_restantes(self):
        from calculations import calcular_velocidade_necessaria
        df = self._make_df([
            {'Data-Lake': 'COMPRAS', 'Status': 'Done',   'Data Atualizacao': '2026-03-01'},
            {'Data-Lake': 'COMPRAS', 'Status': 'Closed', 'Data Atualizacao': '2026-03-01'},
            {'Data-Lake': 'COMPRAS', 'Status': 'To Do',  'Data Atualizacao': '2026-03-01'},
        ])
        lakes = self._make_lakes('COMPRAS', '2026-12-31')
        result = calcular_velocidade_necessaria(df, lakes)
        assert result.loc[result['lake'] == 'COMPRAS', 'itens_restantes'].iloc[0] == 1

    def test_semaforo_risco_when_real_below_50pct(self):
        from calculations import calcular_velocidade_necessaria
        # 100 itens restantes, far deadline -> necessaria muito alta -> ratio < 0.5
        hoje = pd.Timestamp.today().normalize()
        future = hoje + pd.offsets.BusinessDay(10)
        df = self._make_df([
            {'Data-Lake': 'FINANCAS', 'Status': 'To Do', 'Data Atualizacao': str(hoje.date())}
            for _ in range(100)
        ])
        lakes = self._make_lakes('FINANCAS', future.strftime('%Y-%m-%d'))
        result = calcular_velocidade_necessaria(df, lakes)
        # velocidade_real = 0 (none completed recently), ratio = 0 -> risco
        assert result.loc[result['lake'] == 'FINANCAS', 'semaforo'].iloc[0] == 'risco'

    def test_semaforo_ok_when_no_remaining_items(self):
        from calculations import calcular_velocidade_necessaria
        hoje = pd.Timestamp.today().normalize()
        future = hoje + pd.offsets.BusinessDay(10)
        df = self._make_df([
            {'Data-Lake': 'LOGISTICA', 'Status': 'Done', 'Data Atualizacao': str(hoje.date())}
        ])
        lakes = self._make_lakes('LOGISTICA', future.strftime('%Y-%m-%d'))
        result = calcular_velocidade_necessaria(df, lakes)
        # velocidade_necessaria = 0 -> ratio = 1.0 -> ok
        assert result.loc[result['lake'] == 'LOGISTICA', 'semaforo'].iloc[0] == 'ok'

    def test_empty_lakes_returns_empty_dataframe(self):
        from calculations import calcular_velocidade_necessaria
        df = self._make_df([
            {'Data-Lake': 'COMPRAS', 'Status': 'To Do', 'Data Atualizacao': '2026-03-01'},
        ])
        lakes = pd.DataFrame(columns=['lake', 'data_fim'])
        result = calcular_velocidade_necessaria(df, lakes)
        assert result.empty

    def test_deadline_in_past_uses_minimum_1_business_day(self):
        from calculations import calcular_velocidade_necessaria
        df = self._make_df([
            {'Data-Lake': 'FISCAL', 'Status': 'To Do', 'Data Atualizacao': '2026-01-01'},
        ])
        # deadline no passado -> dias_uteis clamped to 1, should not raise ZeroDivision
        lakes = self._make_lakes('FISCAL', '2025-01-01')
        result = calcular_velocidade_necessaria(df, lakes)
        assert result.loc[result['lake'] == 'FISCAL', 'dias_uteis'].iloc[0] == 1
        assert result.loc[result['lake'] == 'FISCAL', 'semaforo'].iloc[0] in {'risco', 'atencao', 'ok'}
```
