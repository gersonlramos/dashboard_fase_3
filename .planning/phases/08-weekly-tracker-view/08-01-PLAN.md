Insert the following block verbatim. Keep the indentation consistent with the other

```python
elif aba_selecionada == "📋 Semanas":
    st.subheader("📋 Semanas — Progresso Semanal por Lake")

    # ── Derived columns ─────────────────────────────────────────────────────────
    _df_sem = _df_semaforos.copy()

    # done_esta_semana: items with status Done updated in the last 7 calendar days
    _janela_7d = _hoje - pd.Timedelta(days=7)
    _status_done_set = {s.lower() for s in _status_done}

    _done_7d_por_lake = {}
    for _lk in _df_sem['lake'].tolist():
        _mask_lake = df['Data-Lake'].astype(str).str.strip().str.upper() == _lk
        _df_lk = df[_mask_lake]
        _mask_done = _df_lk['Status'].astype(str).str.strip().str.lower().isin(_status_done_set)
        _mask_recente = pd.to_datetime(
            _df_lk['Data Atualizacao'], errors='coerce'
        ) >= _janela_7d
        _done_7d_por_lake[_lk] = int((_mask_done & _mask_recente).sum())

    _df_sem['done_esta_semana'] = _df_sem['lake'].map(_done_7d_por_lake).fillna(0).astype(int)

    # semanas_restantes: business days to deadline / 5, floor at 0
    def _semanas_rest(deadline):
        try:
            _bd = int(np.busday_count(_hoje.date(), pd.Timestamp(deadline).date()))
            return round(max(_bd, 0) / 5, 1)
        except Exception:
            return 0.0

    _df_sem['semanas_restantes'] = _df_sem['deadline'].apply(_semanas_rest)

    # semaphore emoji
    _SEMAFORO_EMOJI = {'risco': '🔴', 'atencao': '🟡', 'ok': '🟢'}
    _df_sem['risco_emoji'] = _df_sem['semaforo'].map(_SEMAFORO_EMOJI).fillna('⚪')

    # ── Summary metric cards ────────────────────────────────────────────────────
    _total_restante   = int(_df_sem['itens_restantes'].sum())
    _done_semana_sum  = int(_df_sem['done_esta_semana'].sum())
    _lakes_em_risco   = int((_df_sem['semaforo'] == 'risco').sum())

    _mc1, _mc2, _mc3 = st.columns(3)
    _mc1.metric("📦 Total Restante",       _total_restante)
    _mc2.metric("✅ Done (7 dias)",         _done_semana_sum)
    _mc3.metric("🔴 Lakes em Risco",        _lakes_em_risco)

    st.markdown("---")

    # ── Weekly table ────────────────────────────────────────────────────────────
    _df_display = _df_sem[[
        'lake', 'itens_restantes', 'done_esta_semana',
        'velocidade_real', 'velocidade_necessaria',
        'semanas_restantes', 'risco_emoji',
    ]].rename(columns={
        'lake':                  'Lake',
        'itens_restantes':       'Itens Restantes',
        'done_esta_semana':      'Done (7 dias)',
        'velocidade_real':       'Vel. Real (it/dia)',
        'velocidade_necessaria': 'Vel. Necessária (it/dia)',
        'semanas_restantes':     'Semanas Restantes',
        'risco_emoji':           'Risco',
    })

    # Round float columns for display
    _df_display['Vel. Real (it/dia)']       = _df_display['Vel. Real (it/dia)'].round(2)
    _df_display['Vel. Necessária (it/dia)'] = _df_display['Vel. Necessária (it/dia)'].round(2)

    st.dataframe(_df_display, use_container_width=True, hide_index=True)
```
