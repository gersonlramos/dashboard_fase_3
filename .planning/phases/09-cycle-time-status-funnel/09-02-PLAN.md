**Replace with** (cached call + dual-trace horizontal bar chart + supporting dataframe):

```python
    # --- Ciclo de tempo por fase ---
    st.markdown("### Ciclo de Tempo por Fase (dias úteis)")

    @st.cache_data(ttl=900)
    def _get_ciclo_por_fase(dados_dir, lake):
        return calcular_ciclo_por_fase(dados_dir, lake=lake)

    _df_ciclo = _get_ciclo_por_fase(DADOS_DIR, data_lake_selecionado)

    if _df_ciclo.empty:
        st.info(
            "Historico de transicoes insuficiente para calcular ciclo de tempo. "
            "Verifique se os arquivos historico_completo-*.csv possuem dados de transicao."
        )
    else:
        # Horizontal bar chart: media e mediana por fase
        _fig_ciclo = go.Figure()
        _fig_ciclo.add_trace(go.Bar(
            name="Media (dias uteis)",
            y=_df_ciclo['status'].astype(str).tolist(),
            x=_df_ciclo['media_dias'].tolist(),
            orientation='h',
            marker_color='#4a90d9',
            text=[f"{v:.1f}d" for v in _df_ciclo['media_dias']],
            textposition='outside',
            hovertemplate="<b>%{y}</b><br>Media: %{x:.1f} dias uteis<extra></extra>",
        ))
        _fig_ciclo.add_trace(go.Bar(
            name="Mediana (dias uteis)",
            y=_df_ciclo['status'].astype(str).tolist(),
            x=_df_ciclo['mediana_dias'].tolist(),
            orientation='h',
            marker_color='#52c87a',
            text=[f"{v:.1f}d" for v in _df_ciclo['mediana_dias']],
            textposition='outside',
            hovertemplate="<b>%{y}</b><br>Mediana: %{x:.1f} dias uteis<extra></extra>",
        ))
        _fig_ciclo.update_layout(
            height=400,
            barmode='group',
            title=f"Ciclo de Tempo por Fase — {data_lake_selecionado}",
            xaxis_title="Dias Uteis",
            yaxis_title="Status",
            template=plotly_template,
            paper_bgcolor=plotly_paper_bgcolor,
            plot_bgcolor=plotly_plot_bgcolor,
            font=dict(color=plotly_font_color),
            xaxis=dict(**plotly_axis_style),
            yaxis=dict(
                categoryorder='array',
                categoryarray=['In Test', 'Waiting Test', 'In progress', 'To Do', 'Open'],
                **plotly_axis_style,
            ),
            legend=plotly_legend_style,
        )
        st.plotly_chart(_fig_ciclo, use_container_width=True)

        # Supporting detail table (collapsible)
        with st.expander("Ver tabela detalhada"):
            st.dataframe(
                _df_ciclo.rename(columns={
                    'status': 'Status',
                    'media_dias': 'Media (dias uteis)',
                    'mediana_dias': 'Mediana (dias uteis)',
                    'n_amostras': 'Amostras',
                }),
                use_container_width=True,
                hide_index=True,
            )

        st.caption(
            "Calculado a partir do historico de transicoes de status (historico_completo-*.csv). "
            "Tempo em cada fase = dias uteis entre entrada e saida do status. "
            f"Total de transicoes analisadas: {int(_df_ciclo['n_amostras'].sum())}."
        )
```
