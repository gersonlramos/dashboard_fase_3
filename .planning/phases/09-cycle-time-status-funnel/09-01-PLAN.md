### Task 1: Add calcular_ciclo_por_fase() to calculations.py

Append the following function at the end of the file. It reads all `historico_completo-*.csv` files, optionally filters by lake, computes business-day durations spent in each Status Antigo, and returns a summary DataFrame.

```python
def calcular_ciclo_por_fase(dados_dir, lake=None):
    """
    Le os arquivos historico_completo-*.csv e calcula o tempo medio e mediana
    (em dias uteis) gasto em cada fase de status por subtarefa.
    Retorna DataFrame com colunas: status, media_dias, mediana_dias, n_amostras.
    Filtrado por lake se fornecido (nao None e nao 'Todos').
    """
    historico_dir = os.path.join(dados_dir, 'historico')
    pattern = os.path.join(historico_dir, 'historico_completo-*.csv')
    files = glob.glob(pattern)
    dfs = []
    for f in files:
        try:
            df_h = pd.read_csv(f, encoding='utf-8-sig')
            if df_h.empty:
                continue
            lake_name = os.path.basename(f).replace('historico_completo-', '').replace('.csv', '').upper()
            df_h['lake'] = lake_name
            dfs.append(df_h)
        except Exception:
            continue
    if not dfs:
        return pd.DataFrame(columns=['status', 'media_dias', 'mediana_dias', 'n_amostras'])

    df_all = pd.concat(dfs, ignore_index=True)

    if lake and lake != 'Todos':
        df_all = df_all[df_all['lake'].str.upper() == lake.upper()]

    df_all['Data Mudanca'] = pd.to_datetime(
        df_all['Data Mudanca'], errors='coerce', utc=True
    ).dt.tz_localize(None)
    df_all = df_all.dropna(subset=['Data Mudanca'])
    df_all = df_all.sort_values(['Chave', 'Data Mudanca'])

    records = []
    for chave, grp in df_all.groupby('Chave'):
        grp = grp.sort_values('Data Mudanca').reset_index(drop=True)
        for i in range(len(grp)):
            status = grp.loc[i, 'Status Antigo']
            t_saida = grp.loc[i, 'Data Mudanca']
            if i == 0:
                t_entrada = pd.to_datetime(
                    grp.loc[i, 'Data Criacao'], errors='coerce', utc=True
                )
                if pd.notna(t_entrada):
                    t_entrada = t_entrada.tz_localize(None) if t_entrada.tzinfo else t_entrada
                else:
                    continue
            else:
                t_entrada = grp.loc[i - 1, 'Data Mudanca']

            if pd.isna(t_entrada) or pd.isna(t_saida):
                continue
            dias = np.busday_count(t_entrada.date(), t_saida.date())
            if dias < 0:
                dias = 0
            records.append({'status': status, 'dias': dias, 'chave': chave})

    if not records:
        return pd.DataFrame(columns=['status', 'media_dias', 'mediana_dias', 'n_amostras'])

    df_rec = pd.DataFrame(records)
    flow_order = ['Open', 'To Do', 'In progress', 'Waiting Test', 'In Test']
    df_rec = df_rec[df_rec['status'].isin(flow_order)]

    result = (
        df_rec.groupby('status')['dias']
        .agg(media_dias='mean', mediana_dias='median', n_amostras='count')
        .reset_index()
    )
    result['status'] = pd.Categorical(result['status'], categories=flow_order, ordered=True)
    result = result.sort_values('status').reset_index(drop=True)
    result['media_dias'] = result['media_dias'].round(1)
    result['mediana_dias'] = result['mediana_dias'].round(1)
    return result
```

elif aba_selecionada == "⏱️ Ciclo":
